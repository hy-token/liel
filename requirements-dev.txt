# -*- coding: utf-8 -*-
# PEP 263 encoding declaration so pip reads this file as UTF-8 even on older
# Windows runners where pip may fall back to the locale codec.
#
# Single dependency list for development, tests, documentation builds, and
# release checks. The project keeps this as one file because maintainer
# workflows are intentionally small.
#
# IMPORTANT: these dependencies are maintainer-only. They are not installed by
# `pip install liel` because the project has no `[project.dependencies]` entry
# and keeps the runtime dependency set empty.
#
# Setup:
#   python -m venv .venv
#   .venv\Scripts\activate   # Windows
#   pip install -r requirements-dev.txt
#   maturin develop
#
# To develop or try the MCP server, then run:
#   pip install -e ".[mcp]"
#
# Versions are pinned with a lower bound + compatible-release upper bound so
# that CI reproduces reliably while still picking up patch releases. Bumps are
# intentional (re-run `pip install --upgrade` locally, commit the change).

# --- Extension build ------------------------------------------------------
# maturin: the official PyO3 build tool. It builds the Rust native extension
# (`python/liel/liel.*.pyd/.so`) and installs it into the active environment.
# Use `maturin develop` during local development and `maturin build --release`
# / `maturin sdist` for release checks.
maturin>=1.5,<2.0

# --- Tests ---------------------------------------------------------------
# pytest: runs the Python integration tests under tests/python. Defaults are
# configured in pyproject.toml under [tool.pytest.ini_options].
pytest>=7.4,<9.0

# --- Lint / Format --------------------------------------------------------
# ruff: fast Python linting and formatting. Target directories and enabled
# rules live in pyproject.toml under [tool.ruff]. CI runs the same config with
# `ruff check` and `ruff format --check`.
ruff>=0.5,<1.0

# --- Documentation (MkDocs) ----------------------------------------------
# mkdocs: static site generator for Markdown files under docs/. Use
# `mkdocs serve` for local previews and `mkdocs build --strict` for strict
# link and configuration checks. CI uses the same build command.
mkdocs>=1.5,<2.0

# mkdocs-material: Material Design-based theme used by mkdocs.yml via
# `theme: name: material`.
mkdocs-material>=9.5,<10.0

# --- Release checks -------------------------------------------------------
# twine: official PyPA packaging utility. Before release, run
# `twine check dist/*` to validate wheel and sdist metadata, README rendering,
# classifiers, and required metadata. Publishing is handled by GitHub Actions
# via Trusted Publishing (OIDC), so maintainers should rarely run
# `twine upload` manually.
twine>=5.0,<7.0
